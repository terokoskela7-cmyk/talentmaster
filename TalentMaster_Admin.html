TalentMaster™ — Tietosuojailmoitus ja suostumus

TalentMaster tallentaa pelaajan urheilusuorituksiin liittyviä tietoja jalkapalloseuran valmennustoiminnan kehittämiseksi.

Tallennettavat tiedot:

  • Nimi, syntymäaika, seura ja joukkue
  • Fyysiset testitulokset (nopeus, ketteryys, hyppy)
  • Harjoitteluseurantakirjaukset (SPL 7 kriteeriä)
  • Biologinen ikä ja kasvuvaihetieto ⚠️
  • Pelitaidolliset arvioinnit (ADAR / Game IQ)

Oikeutesi: Voit milloin tahansa tarkastaa, oikaista tai pyytää poistamaan tietosi.

Säilytysaika: Enintään 5 vuotta tai seurajäsenyyden ajan.

Firestore Security Rules
Ajantasaiset säännöt sihteeri-arkkitehtuurilla — kopioi Firebase Consoleen
🛡️firestore.rules
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── Apufunktiot ──────────────────────────────
    function isAuth() { return request.auth != null; }
    function uid() { return request.auth.uid; }

    function isSuperAdmin() {
      return isAuth() &&
        exists(/databases/$(database)/documents/admins/$(uid()));
    }

    // Tarkistaa onko käyttäjällä rooli tietyssä seurassa
    // Roolit on taulukko: roolit: ["vp", "sihteeri"]
    function hasRole(seuraId, rooli) {
      return isAuth() && rooli in
        get(/databases/$(database)/documents/seurat/$(seuraId)/kayttajat/$(uid())).data.roolit;
    }

    function isVP(s)       { return hasRole(s, 'vp'); }
    function isSihteeri(s) { return hasRole(s, 'sihteeri'); }
    function isValmentaja(s){ return hasRole(s, 'valmentaja'); }

    // VP tai Sihteeri — molemmat voivat hallita käyttäjiä
    function canManageUsers(s) {
      return isSuperAdmin() || isVP(s) || isSihteeri(s);
    }

    // ── Super Admin ───────────────────────────────
    match /admins/{docId} {
      allow read, write: if isSuperAdmin();
    }

    // ── Seurat ────────────────────────────────────
    match /seurat/{seuraId} {
      allow read:  if isSuperAdmin() || isVP(seuraId) || isSihteeri(seuraId) || isValmentaja(seuraId);
      allow write: if isSuperAdmin();

      // Joukkueet — Sihteeri ja VP hallinnoi
      match /joukkueet/{joukkueId} {
        allow read:  if isSuperAdmin() || isVP(seuraId) || isSihteeri(seuraId) || isValmentaja(seuraId);
        allow write: if isSuperAdmin() || isSihteeri(seuraId) || isVP(seuraId);
      }

      // Käyttäjät — roolit taulukkona
      match /kayttajat/{kayttajaId} {
        allow read:  if isSuperAdmin() || isVP(seuraId) || isSihteeri(seuraId);
        allow write: if canManageUsers(seuraId);
      }

      // Harjoitteluseurantakirjaukset
      match /kirjaukset/{kirjausId} {
        allow read:  if isSuperAdmin() || isVP(seuraId) || isValmentaja(seuraId);
        allow write: if isSuperAdmin() || isVP(seuraId) || isValmentaja(seuraId);
      }

      // Testitulokset
      match /testit/{testiId} {
        allow read:  if isSuperAdmin() || isVP(seuraId) || isValmentaja(seuraId);
        allow write: if isSuperAdmin() || isVP(seuraId)
                     || hasRole(seuraId,'testi') || hasRole(seuraId,'fysiikka');
      }

      // Harjoitettavuuskartoitukset + Tekniikkakilpailut
      match /kartoitukset/{id} {
        allow read:  if isSuperAdmin() || isVP(seuraId) || isValmentaja(seuraId);
        allow write: if isSuperAdmin() || isVP(seuraId) || isSihteeri(seuraId);
      }
      match /tekniikka/{id} {
        allow read:  if isSuperAdmin() || isVP(seuraId) || isValmentaja(seuraId);
        allow write: if isSuperAdmin() || isVP(seuraId) || isSihteeri(seuraId);
      }

      // Vammadata — arkaluonteinen
      match /vammat/{id} {
        allow read:  if isSuperAdmin() || isVP(seuraId) || hasRole(seuraId,'fysio');
        allow write: if isSuperAdmin() || hasRole(seuraId,'fysio');
      }

      // GDPR-suostumukset
      match /suostumukset/{id} {
        allow read, write: if canManageUsers(seuraId);
      }
    }

    // ── Vanha rakenne (yhteensopivuus) ────────────
    match /kirjaukset/{doc} { allow read, write: if isAuth(); }
    match /kirjaukset_joukkue/{doc} { allow read, write: if isAuth(); }
    match /kirjaukset_tapahtumat/{doc} { allow read, write: if isAuth(); }
  }
}
